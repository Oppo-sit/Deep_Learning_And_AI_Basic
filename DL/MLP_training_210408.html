def ComputeAccr(dloader, imodel):
    correct = 0
    total = 0
    
    for j, [imgs, labels] in enumerate(dloader):
        img = imgs
        # print(img.shape) # (batch size, 1, 28, 28)
        label = Variable(labels) #     Variable   .
        # label = Variable(labels).cuda()
        
        # data    reshape (1, 28, 28) ->(28*28)
        # (batch size, 1, 28, 28) -> (batch size, 28, 28)
        img = img.reshape((img.shape[0], img.shape[2], img.shape[3]))
        # (batch size, 28, 28) -> (batch size, 28*28)
        img = img.reshape((img.shape[0], img.shape[1]*img.shape[2]))
        img = Variable(img, requires_grad=False) # requires_grad :  ?
        # img = Variable(img, requires_grad_False).cuda()
        
        output = imodel(img) # forward propagation #  ''   
        # output shape : (1024, 10), one hot encoding
        _, output_index = torch.max(output, 1) #      .(argmax->yhat)
        # max  : (data, axis) 
        #    => axis   (axis=1  '()' ) shape 1 .
        
        total += label.size(0)
        correct += (output_index == label).sum().float()
        print(total, correct) #        
    
    print("Accuracy of Test Data: {}".format(100*correct/total))
