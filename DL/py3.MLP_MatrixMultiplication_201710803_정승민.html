def sigmoid(x): # activation function : sigmoid
    return 1/(1+np.exp(-x))

def softmax(x): # 분류의 결과값을 확률로, 합을 1로 만들어준다.
    e_x = np.exp(x)
    return e_x/np.sum(e_x)

# Multi Layerd Perceptron 구축
# (256, 785) -> (128, 257) -> (10, 129), 구현과정에서는 bias를 분리한다.
class MyMLP:
    def __init__(self, n_input, n_hidden1, n_hidden2, n_output):# 가중치를 넣을 행렬을 생성자에 추가한다
        self.W1 = np.zeros((n_hidden1, n_input), dtype=np.float32) # (256, 784)
        self.b1 = np.zeros((n_hidden1), dtype=np.float32) # (256, 1)
        
        self.W2 = np.zeros((n_hidden2, n_hidden1), dtype=np.float32) # (128, 256)
        self.b2 = np.zeros((n_hidden2), dtype=np.float32) # (128, 1)
        
        self.W3 = np.zeros((n_output, n_hidden2), dtype=np.float32) # (10, 128)
        self.b3 = np.zeros((n_output), dtype=np.float32) # (10, 1)
        
    def __call__(self, x):
        x = x.reshape(-1) # 이미지를 1차원 array로 만들어준다
        
        # 신경망 연산, activation function을 어떻게 사용할지는 스스로 생각해야 한다.
        h1 = sigmoid(np.dot(self.W1, x) + self.b1) 
        h2 = np.dot(self.W2, h1) + self.b2
        out = np.dot(self.W3, h2) + self.b3
        
        # output을 확률화
        return softmax(out)
